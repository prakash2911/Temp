private List<ThorRequest> addRecordsToList(final List<OdsPosition> positions) {
    return positions.stream()
        .filter(n -> n.getInstrument() != null && n.getInstrument().getUniqueReference() != null)  // Filter out nulls
        .map(n -> new ThorRequest(n.getInstrument().getUniqueReference(), THOR_REQUEST_TYPE))
        .distinct()
        .collect(Collectors.toList());
}
